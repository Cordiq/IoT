Averaging Job

SELECT DateAdd(minute,-5,System.TimeStamp) as StartTime, system.TimeStamp as EndTime, 
Avg(Temp) as AverageTemp, Avg(Humidity) as AverageHumidity, Count(*) as LogCount
    FROM ehinput
    GROUP BY TumblingWindow(minute, 5)
